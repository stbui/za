# Tab

## Installation

```sh
npm install @stbui/za-tab
```

## Usage

```jsx
import React from 'react';
import Tabs from '@stbui/za-tabs';
import Tab from '@stbui/za-tab';

export default () => (
    <>
        <Tabs defaultValue={0}>
            <Tab label="tab1">Tab1</Tab>
            <Tab label="tab2">Tab2</Tab>
            <Tab label="tab3">Tab3</Tab>
            <Tab label="tab4">Tab4</Tab>
        </Tabs>
    </>
);
```
