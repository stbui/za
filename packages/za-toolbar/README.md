# Toolbar

## Installation

```sh
npm install @stbui/za-toolbar
```

## Usage

```jsx
import React from 'react';
import Toolbar from '@stbui/za-toolbar';

export default () => (
    <>
        <Toolbar>Toolbar</Toolbar>
        <Toolbar disableGutters>Toolbar</Toolbar>
        <Toolbar variant="dense">Toolbar</Toolbar>
    </>
);
```
